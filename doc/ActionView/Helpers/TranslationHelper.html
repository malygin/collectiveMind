def t(key, options = {})
  if !@project.nil? and !@project.project_type_id.nil?
    value = I18n.t("#{key}_#{@project.project_type.code}", options)
    return value unless value.match 'translation missing: (.+)'
  end
  I18n.t(key, options)
end
