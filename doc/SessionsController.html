def create
  user = User.authenticate(params[:session][:email], params[:session][:password])
  if user.nil?
    flash.now[:error] = 'Ошибка авторизации!'
    render 'new'
  else
    sign_in user
    journal_enter
    redirect_back_or '/'
  end
  respond_to :js, :html
end
